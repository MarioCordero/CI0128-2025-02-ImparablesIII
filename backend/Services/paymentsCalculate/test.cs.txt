var gross = 3_500_000m;

// Empleado
var empLines = new[]
{
    new Employee.CcssSemEmployeeCalc().Calculate(gross),
    new Employee.CcssIvmEmployeeCalc().Calculate(gross),

{
    new Employer.CcssSemEmployerCalc().Calculate(gross),

    new Employer.InaEmployerCalc().Calculate(gross),
    new Employer.FodesafEmployerCalc().Calculate(gross),
    new Employer.FclEmployerCalc().Calculate(gross),
    new Employer.FpcEmployerCalc().Calculate(gross),
    new Employer.BancoPopularEmployerCalc().Calculate(gross),
};
var patTotal = patLines.Sum(x => x.Amount);
